function loadStudentReports(){ const studentName= document.getElementById("studentName").value.trim(); if(!studentName){ showAlert("اكتب اسم الطالب أولاً","error"); return; } const box= document.getElementById("teacherReportsBox"); box.innerHTML="جاري تحميل التقارير..."; fetch( SCRIPT_URL+ "?action=getStudentReports&studentName="+ encodeURIComponent(studentName) ) .then(r=>r.json()) .then(data=>{ if(!data.length){ box.innerHTML= "لا توجد تقارير لهذا الطالب"; return; } box.innerHTML=""; data.reverse().forEach(report=>{ box.innerHTML+=`

رصد رقم ${report.rowNumber}

الحضور ${report.attendance}
الحفظ/المسار ${report.hifzPart}
معاني الكلمات ${report.maani}
الشواهد ${report.shatibiya}
الملاحظات ${report.notes || "لا توجد"}
`; }); }); } function editReport(report){ editRowNumber= report.rowNumber; document.getElementById("studentName").value= report.studentName || ""; document.getElementById("nationality").value= report.nationality || ""; document.getElementById("studyType").value= report.studyType || ""; document.getElementById("readType").value= report.readType || ""; document.getElementById("attendance").value= report.attendance || ""; document.getElementById("hifzPart").value= report.hifzPart || ""; document.getElementById("tohfa").value= report.tohfa || ""; document.getElementById("jazariya").value= report.jazariya || ""; document.getElementById("maani").value= report.maani || ""; document.getElementById("shatibiya").value= report.shatibiya || ""; document.getElementById("notes").value= report.notes || ""; document.getElementById("editModeBox").style.display="block"; document.getElementById("cancelEditBtn").style.display="block"; window.scrollTo({ top:0, behavior:"smooth" }); } function cancelEdit(clear=true){ editRowNumber=null; document.getElementById("editModeBox").style.display="none"; document.getElementById("cancelEditBtn").style.display="none"; if(clear){ clearForm(); } } function deleteReport(rowNumber){ const confirmDelete= confirm("هل أنت متأكد من حذف هذا الرصد؟"); if(!confirmDelete)return; const img=new Image(); img.onload=img.onerror=function(){ showAlert("تم حذف الرصد بنجاح","success"); loadStudentReports(); }; img.src= SCRIPT_URL+ "?action=delete&rowNumber="+ rowNumber; } function clearForm(){ document.getElementById("studentName").value=""; document.getElementById("nationality").value=""; document.getElementById("notes").value=""; }
